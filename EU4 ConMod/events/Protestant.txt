# The Protestant Reformation
country_event = {
	id = 2001
	title = "EVTNAME2001"
	desc = "EVTDESC2001"
	picture = RELIGION_eventPicture
	
	major = yes
	
	trigger = {
		NOT = { is_religion_enabled = protestant }
		religion = catholic
		culture_group = germanic
		NOT = {
			OR = {
				dominant_culture = austrian
				dominant_culture = bavarian
				dominant_culture = swabian
			}
		}
		is_defender_of_faith = no	
		NOT = { has_reform = papacy_reform }
		reform_desire = 0.95
		capital_scope = {
			continent = europe
		}
		any_owned_province = {
			can_have_center_of_reformation_trigger = {
				RELIGION = protestant
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1000
				
		modifier = {
			factor = 0.75
			reform_desire = 1.0
		}
		modifier = {
			factor = 0.75
			reform_desire = 1.05
		}
		modifier = {
			factor = 0.75
			reform_desire = 1.10
		}
		modifier = {
			factor = 0.66
			reform_desire = 1.15
		}
		modifier = {
			factor = 0.50
			reform_desire = 1.20
		}
		modifier = {
			factor = 0.25
			reform_desire = 1.25
		}
		
		modifier = {
			factor = 0.75
			NOT = { num_of_cities = 10 }
		}
		modifier = {
			factor = 0.75
			NOT = { num_of_cities = 2 }
		}
		
		modifier = {
			factor = 0.75
			has_idea_group = innovativeness_ideas
		}
		modifier = {
			factor = 1.25
			has_idea_group = religious_ideas
		}

		modifier = {
			factor = 0.95
			NOT = { has_opinion = { who = PAP value = -50 } }
		}
		modifier = {
			factor = 0.85
			NOT = { has_opinion = { who = PAP value = -100 } }
		}
		modifier = {
			factor = 0.75
			NOT = { has_opinion = { who = PAP value = -150 } }
		}		
		modifier = {
			factor = 0.5
			num_of_rebel_controlled_provinces = 1
		}
		modifier = {
			factor = 2.0
			government = theocracy
		}
		modifier = {
			factor = 8.0
			OR = {
				culture_group = latin
				culture_group = iberian
			}
		}
	}

	option = {
		name = "EVTOPTA2001"
		enable_religion = protestant
		random_owned_province = {
			limit = {
				can_have_center_of_reformation_trigger = {
					RELIGION = protestant
				}
			}
			change_religion = protestant
			add_reform_center = protestant
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
			if = {
				limit = {
					NOT = { is_capital = yes }
				}
				root = {
					capital_scope = {
						change_religion = protestant
						add_permanent_province_modifier = {
							name = "religious_zeal_at_conv"
							duration = 9000
						}
					}
				}
			}
		}
		set_country_flag = "protestant_reformation" 
	}
}

#Scandanavian Reformation
country_event = {
	id = prot.1
	title = prot.1.t
	desc = prot.1.d
	picture = RELIGION_eventPicture
	
	major = yes
	
	fire_only_once = yes
	
	trigger = {
		capital_scope = {
			region = scandinavia_region
		}
		is_religion_enabled = protestant
		religion = catholic
		any_owned_province = {
			can_have_center_of_reformation_trigger = {
				RELIGION = protestant
			}
		}
	}
	
	mean_time_to_happen = {
		months = 100
	}
	
	option = {		
		name = prot.1.OPTA
		ai_chance = { factor = 0 }
		add_prestige = -100
		add_country_modifier = {
			name = "conversion_zeal"
			duration = 3650
		}	
		add_country_modifier = {
			name = "noble_zeal"
			duration = 3650
		}
		change_religion = protestant
		random_owned_province = {
			limit = {
				can_have_center_of_reformation_trigger = {
					RELIGION = protestant
				}
			}
			change_religion = protestant
			add_reform_center = protestant
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
		}
	}
	
	option = {		
		name = prot.1.OPTB
		ai_chance = { factor = 100 }
		random_owned_province = {
			limit = {
				can_have_center_of_reformation_trigger = {
					RELIGION = protestant
				}
			}
			change_religion = protestant
			add_reform_center = protestant
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
		}
	}
}
